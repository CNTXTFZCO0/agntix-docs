```javascript Node.js
const sessionId = 'ses_01j3m8xkqp8v2nw4f7g9hk0r';

const { data: messages, pagination } = await fetch(
  `https://api.agntix.ai/v1/chat/chat/sessions/${sessionId}/messages?page=1&limit=50`,
  { headers: { 'x-api-key': 'pk_live_••••' } },
).then(r => r.json());

for (const msg of messages) {
  console.log(`[${msg.role}] ${msg.content}`);
}
```

```python Python
import httpx

session_id = "ses_01j3m8xkqp8v2nw4f7g9hk0r"

result = httpx.get(
    f"https://api.agntix.ai/v1/chat/chat/sessions/{session_id}/messages",
    headers={"x-api-key": "pk_live_••••"},
    params={"page": 1, "limit": 50},
).json()

for msg in result["data"]:
    print(f"[{msg['role']}] {msg['content']}")
```
